34 
35  for( kuiIndex = 0; kuiIndex < auiBytes; kuiIndex++ )
36  {
37  // 1. XOR the input byte with the low-order byte of the CRC register to get an INDEX
38  // 2. Shift the CRC register eight bits to the right
39  // 3. XOR the CRC register with the contents of Table[ INDEX ]
40  // 4. Repeat steps 1 through 3 for all bytes
41  kuiValue = auiTable[ *kucpByte++ ^ ( kuiValue & xuiMaskByte ) ] ^ ( kuiValue >> xuiShift8 );
